# make a new project
new-project 400 400
# add a layer
add-layer first-layer
# add the image to the layer
add-image-to-layer first-layer res/logo.ppm 20 20
# set the filter as green-component
set-filter first-layer green-component
# save image as greenLogo.ppm
save-image res/greenLogo.ppm
# save the project
save-project res/greenLogo.collage
# quit
quit
