# make a new project
new-project 400 400
# add a layer
add-layer first-layer
# add the image to the layer
add-image-to-layer first-layer res/logo.ppm 20 20
# set the filter as red-component
set-filter first-layer red-component
# save the project
save-project res/redLogo.collage
# save image as redLogo.ppm
save-image res/redLogo.ppm
# quit
quit
